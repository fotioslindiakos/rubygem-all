      def has_hook_method(options)
        source = []
        options.each_pair do |name, event|
          source << "def \#{name}(&blk)\non_hook \#{event.to_sym.inspect}, &blk\nend\n"
        end
        class_eval source.join("\n"), __FILE__, __LINE__
      end
