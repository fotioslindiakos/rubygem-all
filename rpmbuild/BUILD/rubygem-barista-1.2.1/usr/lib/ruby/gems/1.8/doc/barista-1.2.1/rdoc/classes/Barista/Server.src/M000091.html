    def forbidden
      @_forbidden_response ||= begin
        body = "Forbidden\n"
        [403, {
          'Content-Type'   => 'text/plain',
          'Content-Length' => Rack::Utils.bytesize(body).to_s,
          'X-Cascade'      => 'pass'
        }, [body]]
      end
    end
