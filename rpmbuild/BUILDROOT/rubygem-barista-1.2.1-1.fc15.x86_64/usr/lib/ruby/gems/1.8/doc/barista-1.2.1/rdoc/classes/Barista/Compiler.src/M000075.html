    def save(path = @options[:output_path])
      return false unless path.is_a?(String) && !to_js.nil?
      FileUtils.mkdir_p File.dirname(path)
      File.open(path, "w+") { |f| f.write @compiled_content }
      true
    rescue Errno::EACCES
      false
    end
