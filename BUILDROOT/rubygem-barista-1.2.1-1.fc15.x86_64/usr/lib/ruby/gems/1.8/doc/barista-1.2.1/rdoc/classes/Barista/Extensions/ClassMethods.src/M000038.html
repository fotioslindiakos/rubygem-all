      def has_boolean_options(*names)
        source = []
        names.each do |name|
          source << "\ndef \#{name}!\n@\#{name} = true\nend\n\ndef \#{name}?\ndefined?(@\#{name}) ? @\#{name} : default_for_\#{name}\nend\n\ndef \#{name}=(value)\n@\#{name} = !!value\nend\n\ndef default_for_\#{name}\nfalse\nend\n\n"
        end
        class_eval source.join("\n"), __FILE__, __LINE__
      end
