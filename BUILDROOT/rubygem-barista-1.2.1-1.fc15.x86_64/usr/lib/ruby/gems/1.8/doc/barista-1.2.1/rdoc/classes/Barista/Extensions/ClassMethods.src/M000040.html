      def has_delegate_methods(delegate, *args)
        source = []
        args.each do |method|
          source << "\ndef \#{method}(*args, &blk)\n\#{delegate}.send(:\#{method}, *args, &blk)\nend\n\n"
        end
        class_eval source.join("\n"), __FILE__, __LINE__
      end
