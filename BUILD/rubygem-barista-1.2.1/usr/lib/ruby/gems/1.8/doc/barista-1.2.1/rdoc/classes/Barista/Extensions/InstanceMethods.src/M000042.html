      def deprecate!(object, method, other = nil)
        klass_prefix = (object.is_a?(Class) || object.is_a?(Module)) ? "#{object.name}." : "#{object.class.name}#"
        warn "#{klass_prefix}#{method} is deprecated and will be removed in 1.0. #{other}".strip
      end
