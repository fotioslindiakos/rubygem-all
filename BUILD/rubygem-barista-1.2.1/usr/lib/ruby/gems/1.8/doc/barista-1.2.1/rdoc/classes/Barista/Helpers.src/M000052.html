    def normalise_coffeescript_path(path)
      if respond_to?(:compute_public_path)
        compute_public_path path, 'coffeescript', 'coffee'
      else
        path = path.gsub(/\.(js|coffee)$/, '') + '.coffee'
        path = "/coffeescripts/#{path}" unless path =~ /^\//
        path
      end
    end
